import tkinter as tk

def show_frame(frame):
    frame.tkraise()

root = tk.Tk()
root.title("Screen Change Example")

container = tk.Frame(root)
container.pack(side="top", fill="both", expand=True)
container.grid_rowconfigure(0, weight=1)
container.grid_columnconfigure(0, weight=1)

frames = {}
for page_name in ("StartPage", "PageOne", "PageTwo"):
    frame = tk.Frame(container)
    frames[page_name] = frame
    frame.grid(row=0, column=0, sticky="nsew")

# StartPage
label_start = tk.Label(frames["StartPage"], text="This is the start page")
label_start.pack(pady=10, padx=10)
button1 = tk.Button(frames["StartPage"], text="Go to Page One", command=lambda: show_frame(frames["PageOne"]))
button1.pack()
button2 = tk.Button(frames["StartPage"], text="Go to Page Two", command=lambda: show_frame(frames["PageTwo"]))
button2.pack()

# PageOne
label_page1 = tk.Label(frames["PageOne"], text="This is Page One")
label_page1.pack(pady=10, padx=10)
button_page1 = tk.Button(frames["PageOne"], text="Go to Start Page", command=lambda: show_frame(frames["StartPage"]))
button_page1.pack()

# PageTwo
label_page2 = tk.Label(frames["PageTwo"], text="This is Page Two")
label_page2.pack(pady=10, padx=10)
button_page2 = tk.Button(frames["PageTwo"], text="Go to Start Page", command=lambda: show_frame(frames["StartPage"]))
button_page2.pack()

show_frame(frames["StartPage"])

root.mainloop()
